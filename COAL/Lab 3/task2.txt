TITLE Add and Subtract (AddSub.asm)
; This program adds and subtracts 32-bit integers.
INCLUDE Irvine32.inc

.data
val1 WORD ?
val2 SBYTE -11
val3 SDWORD -2147483648 
.code

main PROC
mov ax, val1
mov ebx, 0
mov bl, val2

; task 2
mov ecx, val3

call DumpRegs ; display registers
exit
main ENDP
END main
