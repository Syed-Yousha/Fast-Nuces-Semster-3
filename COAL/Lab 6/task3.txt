TITLE My First Program (Test.asm)
INCLUDE Irvine32.inc

.data
array1 dword 1,2,3,4,5
array2 dword 6,7,8,9,10
promt byte "Sum of arrays is: ",0

.code
main PROC

call array1_sum
call array2_sum
call add_sums

mov edx, offset promt
mov eax, ecx

call WriteString
call WriteDec
call crlf

CALL DumpRegs
EXIT
main ENDP

array1_sum PROC USES esi ecx

	mov esi, offset array1
	mov eax, 0
	mov ecx, lengthof array1

	l1:
		add eax, [esi]
		add esi, type array1
	loop l1

	ret 
array1_sum ENDP

array2_sum PROC USES edi ecx

	mov edi, offset array2
	mov ebx, 0
	mov ecx, lengthof array2

	l1:
		add ebx, [edi]
		add edi, type array2
	loop l1

	ret 
array2_sum ENDP

add_sums PROC USES eax ebx
	mov ecx, eax 
	add ecx, ebx
	
	ret
add_sums ENDP

END main
