TITLE My First Program (Test.asm)
INCLUDE Irvine32.inc

.data
promt byte "Enter any number: "

.code
main PROC

call num_sum

CALL DumpRegs
EXIT
main ENDP

num_sum PROC

	mov edx, offset promt
	call WriteString
	call ReadDec

	mov ebx, 0
	mov ecx, eax
	
	l1:
		
		add ebx, ecx

	loop l1

	mov eax, ebx
	call crlf
	call WriteDec
	call crlf

	ret
num_sum ENDP

END main
