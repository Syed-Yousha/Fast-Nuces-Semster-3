INCLUDE Irvine32.inc

.data
    target BYTE "AAEBDCFBBC", 0
    freqTable DWORD 256 DUP(0)

.code
main PROC
    INVOKE Get_frequencies, ADDR target, ADDR freqTable
    exit
main ENDP

Get_frequencies PROC
    mov  esi, 0

    find_end:
        mov  al, [edx + esi]
        cmp  al, 0
        je   done_frequency
        inc  freqTable[al]
        inc  esi
        jmp  find_end

    done_frequency:
        ret
Get_frequencies ENDP

END main
