INCLUDE Irvine32.inc

.data
    Array DWORD 10, 20, 30, 40, 50
    ArraySize DWORD 5
    Multiplier BYTE 3

.code
main PROC
    call MultiplyArray
    call PrintIntArray
    exit
main ENDP

MultiplyArray PROC
    mov  esi, OFFSET Array
    mov  ecx, ArraySize
    mov ebx, 0
    mov  bl, Multiplier
    multiply_loop:
        mov  eax, [esi]
        imul eax, ebx
        mov  [esi], eax
        add  esi, 4
        loop multiply_loop
    ret
MultiplyArray ENDP

PrintIntArray PROC
    mov  esi, OFFSET Array
    mov  ecx, ArraySize
    print_loop:
        mov  eax, [esi]
        call WriteDec
        call Crlf
        add  esi, 4
        loop print_loop
    ret
PrintIntArray ENDP

END main
