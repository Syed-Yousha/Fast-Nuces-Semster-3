TITLE My First Program (Test.asm)
INCLUDE Irvine32.inc

.data
tmp dword 0
itr dword 5
array dword 4,3,2,1

.code
main PROC

mov ecx, 4
mov esi, 0

L1:
	mov tmp, ecx
	dec itr
	mov ecx, itr
	L2:
		mov eax, array[esi]
		call WriteDec
		add esi, TYPE array

	loop L2
	mov esi, 0

	call crlf
	mov ecx, tmp

loop L1

call DumpRegs
exit

main ENDP
END main
