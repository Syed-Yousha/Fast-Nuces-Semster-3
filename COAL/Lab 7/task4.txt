INCLUDE Irvine32.inc

.data
    numRows DWORD 5
    numCols DWORD ?

.code
main PROC
    ; Get the number of columns from the user
    mov edx, OFFSET prompt
    call WriteString
    call ReadInt
    mov numCols, eax

    ; Call the function to print the pattern
    mov ecx, numRows
    call PrintPattern

    exit
main ENDP

PrintPattern PROC
    ; Parameters: ECX = number of rows, numCols = number of columns
    mov  eax, 1

    outer_loop:
        mov  ebx, numCols
        call PrintSpaces

        mov  ebx, eax
        call PrintStars

        call Crlf

        inc  eax
        loop outer_loop

    ret
PrintPattern ENDP

PrintSpaces PROC
    ; Parameters: EBX = number of spaces to print
    spaces_loop:
        call WriteChar
        dec  ebx
        jnz  spaces_loop
    ret
PrintSpaces ENDP

PrintStars PROC
    ; Parameters: EBX = number of stars to print
    stars_loop:
        call WriteChar
        dec  ebx
        jnz  stars_loop
    ret
PrintStars ENDP

prompt BYTE "Enter the number of columns: ", 0

END main
