INCLUDE Irvine32.inc

.data
    numSum DWORD ?

.code
main PROC
    mov edx, OFFSET prompt
    call WriteString
    call ReadInt
    mov numSum, eax

    mov eax, numSum
    call SumNumbers
    mov edx, OFFSET resultMsg
    call WriteString
    call WriteInt
    call Crlf

    exit
main ENDP

SumNumbers PROC
    mov ecx, eax
    mov eax, 0

    sum_loop:
        add eax, ecx
        loop sum_loop
    ret
SumNumbers ENDP

prompt BYTE "Enter a number: ", 0
resultMsg BYTE "Sum of numbers: ", 0

END main
